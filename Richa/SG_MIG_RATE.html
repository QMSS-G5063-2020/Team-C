# List of iso3 country codes
need <- c("SSD", "SYR", "VEN")

# GDP Growth Rate
mig_data <- wb(country = need, indicator = "SM.POP.NETM", startdate = 2000, enddate = 2020)
ggplot(mig_data, aes(x = as.factor(date), y = value, color = country)) +
  geom_line(aes(group = 1)) + geom_point() + facet_grid(rows = vars(country)) +
  labs(title = "Migration Flow",
       subtitle = "changes in migration in last 20 years", parse = TRUE) +
  xlab(NULL) + ylab(NULL)
